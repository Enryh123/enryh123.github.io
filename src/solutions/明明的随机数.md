```cpp title="参考程序"
int n, x, b[1001];
cin>>n; // 输入10，代表有10个数要排序
for(int i=1; i<=n; i++){
	cin>>x; // 输入20
	b[x]++; // 20出现了1次
}
int cnt=0; // 统计不重复的个数
/*
if(b[1]>0) cnt++; // 1出现过
if(b[2]>0) cnt++; // 2出现过
if(b[3]>0) cnt++; // 3出现过
...
if(b[1000]>0) cnt++; // 1000出现过
*/
for(int i=1; i<=1000; i++) if(b[i]>0) cnt++;
cout<<cnt<<endl; // 不重复的个数
/*
if(b[1]>0) cout<<1<<' ';
if(b[2]>0) cout<<2<<' ';
...
if(b[1000]>0) cout<<1000<<' ';
*/
for(int i=1; i<=1000; i++) if(b[i]>0) cout<<i<<' '; // 从小到大排序
```
